# CUDA-Implementation-of-GPU-vs-GPU-Dice-Battle-Game
### Project Overview
This project demonstrates a simple GPU vs GPU Dice Battle game using CUDA. Two GPU players compete against each other by automatically rolling dice using CUDA kernels. The program compares the dice values generated by both GPU players and determines the winner for each round.

The project is designed to demonstrate CUDA programming, GPU computation, and basic parallel processing concepts through a simple gaming application. Multiple rounds are executed automatically, and the results are displayed in the terminal showing the dice values generated by both GPU players and the final winner of each round.

### Features
1.GPU vs GPU gameplay

2.CUDA kernel execution

3.Automatic dice roll generation

4.Multiple round execution

5.Terminal-based output visualization

### PROGRAM:
```

%%writefile gpu_dice_battle.cu

#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <cuda.h>

__global__ void rollDice(int *dice, int seed)
{
    int idx = threadIdx.x;
    dice[idx] = ((seed + idx * 7) % 6) + 1;
}

void findWinner(int p1, int p2)
{
    printf("GPU 1 rolled: %d\n", p1);
    printf("GPU 2 rolled: %d\n", p2);

    if(p1 > p2)
        printf("Winner: GPU 1\n");

    else if(p2 > p1)
        printf("Winner: GPU 2\n");

    else
        printf("Result: Draw\n");
}

int main()
{
    int *d_dice;
    int h_dice[2];

    cudaMalloc((void**)&d_dice, 2 * sizeof(int));

    srand(time(0));

    for(int round = 1; round <= 5; round++)
    {
        printf("\nRound %d\n", round);

        int seed = rand();

        rollDice<<<1,2>>>(d_dice, seed);

        cudaMemcpy(h_dice, d_dice, 2*sizeof(int), cudaMemcpyDeviceToHost);

        findWinner(h_dice[0], h_dice[1]);
    }

    cudaFree(d_dice);

    return 0;
}
```
